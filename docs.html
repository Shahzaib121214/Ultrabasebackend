import retrofit2.Response import retrofit2.http.* interface UltraBaseAPI { @POST("api.php") suspend fun saveData( @Body request: SaveRequest ): Response<SaveResponse> @GET("api.php") suspend fun getData( @Query("action") action: String = "get", @Query("key") key: String, @Query("pid") projectId: Int, @Query("collection") collection: String ): Response<List<DataResponse>> @POST("api.php") suspend fun login( @Body request: LoginRequest ): Response<AuthResponse> } // Data Classes data class SaveRequest( val action: String = "save", val key: String, val pid: Int, val collection: String, val data: Any ) data class SaveResponse( val status: String, val id: Int ) data class DataResponse( val db_id: Int, val data: Map<String, Any> ) data class LoginRequest( val action: String = "login", val key: String, val username: String, val password: String ) data class AuthResponse( val status: String, val message: String, val user: User? ) data class User( val id: Int, val username: String, val email: String )

Retrofit Instance Setup

Ek singleton object banayein: RetrofitClient.kt

import retrofit2.Retrofit import retrofit2.converter.gson.GsonConverterFactory object RetrofitClient { // Apna UltraBase URL yahan dalein private const val BASE_URL = "http://your-domain.com/htdocs/" // Apni secret key yahan dalein const val API_KEY = "maxclube_secret" // Apna project ID yahan dalein const val PROJECT_ID = 1 val api: UltraBaseAPI by lazy { Retrofit.Builder() .baseUrl(BASE_URL) .addConverterFactory(GsonConverterFactory.create()) .build() .create(UltraBaseAPI::class.java) } }
5

Repository Class Banayein

UltraBaseRepository.kt

class UltraBaseRepository { private val api = RetrofitClient.api // Data Save Karein suspend fun saveData(collection: String, data: Any): SaveResponse? { val request = SaveRequest( key = RetrofitClient.API_KEY, pid = RetrofitClient.PROJECT_ID, collection = collection, data = data ) val response = api.saveData(request) return if (response.isSuccessful) { response.body() } else { null } } // Data Get Karein suspend fun getData(collection: String): Map<String, Any>? { val response = api.getData( key = RetrofitClient.API_KEY, projectId = RetrofitClient.PROJECT_ID, collection = collection ) return if (response.isSuccessful && response.body()?.isNotEmpty() == true) { response.body()?.get(0)?.data } else { null } } // Login suspend fun login(username: String, password: String): AuthResponse? { val request = LoginRequest( key = RetrofitClient.API_KEY, username = username, password = password ) val response = api.login(request) return if (response.isSuccessful) { response.body() } else { null } } }

Authentication Endpoints

Action Method Parameters Description
register POST username, email, password New user registration
login POST username, password User login
logout POST - User logout
check_session GET - Check if user is logged in

Data Management Endpoints

Action Method Parameters Description
save POST pid, collection, data Save/update data
get GET pid, collection Retrieve data
search_data GET pid, query Search across collections

Project Management

Action Method Parameters Description
get_projects GET - Get user's projects
create_project POST name Create new project
delete_project GET id Delete project

Advanced Features

Action Method Parameters Description
