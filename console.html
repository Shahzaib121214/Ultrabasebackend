// ViewModel mein
fun saveUserData(name: String, email: String) {
    viewModelScope.launch {
        try {
            val userData = mapOf(
                "name" to name,
                "email" to email,
                "timestamp" to System.currentTimeMillis()
            )
            
            val response = RetrofitClient.api.saveData(
                SaveRequest(
                    action = "save",
                    key = RetrofitClient.API_KEY,
                    pid = RetrofitClient.PROJECT_ID,
                    collection = "users",
                    data = userData
                )
            )
            
            if (response.isSuccessful) {
                println("‚úÖ Data saved!")
            }
        } catch (e: Exception) {
            println("‚ùå Error: ${e.message}")
        }
    }
}

// ViewModel mein
fun getUserData() {
    viewModelScope.launch {
        try {
            val response = RetrofitClient.api.getData(
                action = "get",
                key = RetrofitClient.API_KEY,
                projectId = RetrofitClient.PROJECT_ID,
                collection = "users"
            )
            
            if (response.isSuccessful) {
                val data = response.body()?.get(0)?.data
                val name = data?.get("name") as? String
                val email = data?.get("email") as? String
                println("‚úÖ User: $name, Email: $email")
            }
        } catch (e: Exception) {
            println("‚ùå Error: ${e.message}")
        }
    }
}
